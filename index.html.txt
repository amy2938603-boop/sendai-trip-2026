        function calculateNetBalances(expenses) {
            const netBalances = {};
            PAYER_OPTIONS.forEach(m => netBalances[m] = { TWD: 0, JPY: 0 });

            expenses.forEach(expense => {
                const currency = expense.currency;
                const amount = expense.amount;
                const sharedBy = expense.sharedBy && Array.isArray(expense.sharedBy) && expense.sharedBy.length > 0 ? expense.sharedBy : PAYER_OPTIONS; 
                const costPerPerson = amount / sharedBy.length;

                if (netBalances[expense.payerName]) netBalances[expense.payerName][currency] += amount;
                sharedBy.forEach(member => {
                    if (netBalances[member]) netBalances[member][currency] -= costPerPerson;
                });
            });
            return netBalances;
        }

        function minimizeTransactions(netBalances, currency) {
            let creditors = [], debtors = []; 
            const tolerance = 0.5;
            
            PAYER_OPTIONS.forEach(name => {
                const balance = netBalances[name][currency];
                if (balance > tolerance) creditors.push({ name, amount: balance });
                else if (balance < -tolerance) debtors.push({ name, amount: -balance });
            });
            
            const settlements = [];
            let i = 0, j = 0;
            debtors.sort((a, b) => b.amount - a.amount);
            creditors.sort((a, b) => b.amount - a.amount);

            while (i < debtors.length && j < creditors.length) {
                const amount = Math.min(debtors[i].amount, creditors[j].amount);
                if (amount > tolerance) { 
                    settlements.push({ from: debtors[i].name, to: creditors[j].name, amount, currency });
                    debtors[i].amount -= amount;
                    creditors[j].amount -= amount;
                }
                if (debtors[i].amount <= tolerance) i++;
                if (creditors[j].amount <= tolerance) j++;
            }
            return settlements;
        }

        function renderSettlement(expenses) {
            const netBalancesEl = document.getElementById('settlement-net-balances');
            const transactionsEl = document.getElementById('settlement-transactions');

                 return;
            }

            const netBalances = calculateNetBalances(expenses);
            const allSettlements = [...minimizeTransactions(netBalances, 'TWD'), ...minimizeTransactions(netBalances, 'JPY')];

            let balanceHtml = '';
            PAYER_OPTIONS.forEach(name => {
                const twd = netBalances[name].TWD || 0;
                const jpy = netBalances[name].JPY || 0;
                const totalEq = twd + convertToTWD(jpy, 'JPY');
                const boxColor = totalEq >= 0 ? 'bg-green-50 border-green-300' : 'bg-red-50 border-red-300';
                
                balanceHtml += `
                    <div class="p-3 rounded-xl shadow-inner ${boxColor} border-l-4">
                        <span class="font-bold text-lg block mb-1 text-gray-800">${name}</span>
                        <div class="flex justify-between text-sm">
                            <span class="text-gray-600">台幣</span>
                            <span class="${twd >= 0 ? 'text-green-600' : 'text-red-600'} font-mono">${twd >= 0 ? '盈餘' : '欠款'}: $${Math.abs(twd).toFixed(0)}</span>
                        </div>
                        <div class="flex justify-between text-sm">
                            <span class="text-gray-600">日圓</span>
                            <span class="${jpy >= 0 ? 'text-green-600' : 'text-red-600'} font-mono">${jpy >= 0 ? '盈餘' : '欠款'}: ￥${Math.abs(jpy).toFixed(0)}</span>
                        </div>
                    </div>`;
            });
            netBalancesEl.innerHTML = balanceHtml;

            let transactionHtml = '';
            if (allSettlements.length === 0) {
                 transactionHtml = '<div class="text-center p-4 text-green-600 border border-green-200 rounded">✨ 結清！大家已經扯平了。</div>';
            } else {
                transactionHtml += '<ul class="space-y-4">';
                allSettlements.forEach(t => {
                    const symbol = t.currency === 'JPY' ? '￥' : '$';
                    const color = t.currency === 'JPY' ? 'text-yellow-700' : 'text-primary-red-bean';
                    const bg = t.currency === 'JPY' ? 'bg-yellow-50' : 'bg-red-50';
                    const icon = t.currency === 'JPY' ? 'yen-square' : 'dollar-sign';
                    transactionHtml += `
                        <li class="p-4 ${bg} rounded-xl snow-card-shadow flex items-center justify-between border-l-4 border-red-400">
                            <span class="font-medium text-gray-700 w-1/5">${t.from}</span>
                            <div class="flex items-center"><i data-lucide="arrow-right" class="w-5 h-5 text-gray-400"></i></div>
                            <div class="text-center flex-1">
                                <p class="text-xs text-gray-500 flex justify-center mb-0.5"><i data-lucide="${icon}" class="w-3 h-3 mr-1"></i>${t.currency} 結算</p>
                                <p class="text-2xl font-extrabold ${color}">${symbol}${t.amount.toFixed(0)}</p>
                            </div>
                            <div class="flex items-center"><i data-lucide="arrow-right" class="w-5 h-5 text-gray-400"></i></div>
                            <span class="font-medium text-gray-700 w-1/5 text-right">${t.to}</span>
                        </li>`;
                });
                transactionHtml += '</ul>';
            }
            transactionsEl.innerHTML = transactionHtml;
            lucide.createIcons();
        }

        function renderExpenseList() {
            if (!isAuthReady) return;
            const loadingEl = document.getElementById('loading-indicator');
            const listEl = document.getElementById('expense-list-container');
            
            onSnapshot(query(getExpensesCollectionRef()), (snapshot) => {
                loadingEl.style.display = 'none';
                let totalTWD = 0;
                const expensesByDate = {};
                const allExpenses = [];

                if (snapshot.empty) {
                    listEl.innerHTML = '<div class="text-center p-8 text-gray-500 bg-white rounded-xl shadow-lg">目前沒有記錄。點擊右上角「新增」！</div>';
                    document.getElementById('total-amount').textContent = 'TWD $0';
                    renderSettlement([]);
                    return;
                }

                snapshot.forEach(doc => {
                    const data = doc.data();
                    const expense = { id: doc.id, ...data };
                    allExpenses.push(expense);
                    totalTWD += convertToTWD(expense.amount, expense.currency);
                    
                    const dateKey = expense.expenseDate || 'Unknown';
                    if (!expensesByDate[dateKey]) expensesByDate[dateKey] = [];
                    expensesByDate[dateKey].push(expense);
                });
                
                renderSettlement(allExpenses);
                document.getElementById('total-amount').textContent = `TWD $${totalTWD.toFixed(0).toLocaleString('en-US')}`;

                let html = '';
                Object.keys(expensesByDate).sort().forEach((dateKey, idx) => {
                    const dayIndex = dateKey !== 'Unknown' ? getDayIndex(dateKey) : '?';
                    const mt = idx === 0 ? '' : 'mt-8';
                    html += `<div class="day-sticky-header bg-white p-3 rounded-lg shadow-sm border-l-4 border-accent-light ${mt} z-5">
                        <h3 class="text-xl font-bold text-gray-800">第 ${dayIndex} 天 - ${dateKey.slice(5).replace('-','/')}</h3></div>
                        <div class="space-y-3 mt-3 mb-8">`;
                    
                    expensesByDate[dateKey].sort((a,b) => (b.dateAdded?.toMillis() || 0) - (a.dateAdded?.toMillis() || 0)).forEach(exp => {
                        const isSelf = exp.recordedByUserId === userId;
                        const symbol = exp.currency === 'JPY' ? '￥' : '$';
                        const sharedNames = (exp.sharedBy || PAYER_OPTIONS).join('、');
                        
                        html += `
                            <div id="expense-${exp.id}" class="bg-white p-4 rounded-xl snow-card-shadow border-l-4 ${isSelf ? 'border-green-500' : 'border-red-500'} flex justify-between border border-red-100">
                                <div class="flex-1 min-w-0">
                                    <p class="text-lg font-medium text-gray-900 truncate">${exp.description}</p>
                                    <div class="text-sm mt-2 space-y-1">
                                        <p class="font-bold text-primary-red-bean">付款: ${exp.payerName}</p>
                                        <p class="text-gray-700">分擔: <span class="text-blue-600">${sharedNames}</span></p>
                                    </div>
                                    <div class="text-xs mt-3 ${isSelf ? 'bg-green-100 text-green-700' : 'bg-red-100 text-red-700'} px-2 py-0.5 rounded inline-block">ID: ${exp.recordedByUserId}</div>
                                </div>
                                <div class="text-right ml-4">
                                    <p class="text-xl font-extrabold text-primary-red-bean">${exp.currency} ${symbol}${exp.amount.toLocaleString()}</p>
                                    ${isSelf ? `<button onclick="deleteExpense('${exp.id}')" class="text-xs text-red-400 hover:text-red-600 mt-1 ml-auto flex items-center"><i data-lucide="trash-2" class="w-3 h-3 mr-1"></i>刪除</button>` : ''}
                                </div>
                            </div>`;
                    });
                    html += `</div>`;
                });
                listEl.innerHTML = html;
                lucide.createIcons();
            });
        }
